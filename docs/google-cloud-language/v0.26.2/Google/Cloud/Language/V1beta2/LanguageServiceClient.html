  end

    "language_service_client_config.json"
  )

    Google::Gax.construct_settings(
      "google.cloud.language.v1beta2.LanguageService",
      JSON.parse(f.read),
      client_config,
      Google::Gax::Grpc::STATUS_CODE_NAMES,
      timeout,
      errors: Google::Gax::Grpc::API_ERRORS,
      kwargs: headers
    )
  end
