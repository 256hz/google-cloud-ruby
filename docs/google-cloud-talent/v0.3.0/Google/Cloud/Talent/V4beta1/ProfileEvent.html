#jobs ⇒ Array<String>

Returns Optional.

The job ID associated with this client event if there is one. Leave it empty if the event isn't associated with a job.

class ProfileEvent
  # The enum represents types of client events for a candidate profile.
  module ProfileEventType
    # Default value.
    PROFILE_EVENT_TYPE_UNSPECIFIED = 0

    # The profile is displayed.
    IMPRESSION = 1

    # The profile is viewed.
    VIEW = 2

    # The profile is bookmarked.
    BOOKMARK = 3
  end
end
